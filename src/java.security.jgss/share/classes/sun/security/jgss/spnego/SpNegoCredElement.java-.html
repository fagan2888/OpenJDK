  25 package sun.security.jgss.spnego;
  26 
  27 import org.ietf.jgss.*;
  28 import java.security.Provider;
  29 import sun.security.jgss.GSSUtil;
  30 import sun.security.jgss.spi.GSSNameSpi;
  31 import sun.security.jgss.spi.GSSCredentialSpi;
  32 
  33 /**
  34  * This class is the cred element implementation for SPNEGO mech.
  35  * NOTE: The current implementation can only support one mechanism.
  36  * This should be changed once multi-mechanism support is needed.
  37  *
  38  * @author Valerie Peng
  39  * @since 1.6
  40  */
  41 public class SpNegoCredElement implements GSSCredentialSpi {
  42 
  43     private GSSCredentialSpi cred = null;
  44 
  45     public SpNegoCredElement(GSSCredentialSpi cred) throws GSSException {
  46         this.cred = cred;
  47     }
  48 
  49     Oid getInternalMech() {
  50         return cred.getMechanism();
  51     }
  52 
  53     // Used by GSSUtil.populateCredentials()
  54     public GSSCredentialSpi getInternalCred() {
  55         return cred;
  56     }
  57 
  58     public Provider getProvider() {
  59         return SpNegoMechFactory.PROVIDER;
  60     }
  61 
  62     public void dispose() throws GSSException {
  63         cred.dispose();
  64     }
  65 
  66     public GSSNameSpi getName() throws GSSException {
  67         return cred.getName();
  68     }
  69 
  70     public int getInitLifetime() throws GSSException {
  71         return cred.getInitLifetime();
  72     }
  73 
  74     public int getAcceptLifetime() throws GSSException {
  75         return cred.getAcceptLifetime();
  76     }
  77 
  78     public boolean isInitiatorCredential() throws GSSException {
  79         return cred.isInitiatorCredential();
  80     }
  81 
  82     public boolean isAcceptorCredential() throws GSSException {
  83         return cred.isAcceptorCredential();
  84     }
  85 
  86     public Oid getMechanism() {
  87         return GSSUtil.GSS_SPNEGO_MECH_OID;
  88     }
  89 
  90     @Override
  91     public GSSCredentialSpi impersonate(GSSNameSpi name) throws GSSException {
  92         return cred.impersonate(name);
  93     }
  94 }
