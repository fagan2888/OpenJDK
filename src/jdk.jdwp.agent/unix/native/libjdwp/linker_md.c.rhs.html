  45 static void dll_build_name(char* buffer, size_t buflen,
  46                            const char* paths, const char* fname) {
  47     char *path, *paths_copy, *next_token;
  48 
  49     paths_copy = strdup(paths);
  50     if (paths_copy == NULL) {
  51         return;
  52     }
  53 
  54     next_token = NULL;
  55     path = strtok_r(paths_copy, PATH_SEPARATOR, &next_token);
  56 
  57     while (path != NULL) {
  58         snprintf(buffer, buflen, "%s/lib%s." LIB_SUFFIX, path, fname);
  59         if (access(buffer, F_OK) == 0) {
  60             break;
  61         }
  62         *buffer = '\0';
  63         path = strtok_r(NULL, PATH_SEPARATOR, &next_token);
  64     }
  65 
  66     free(paths_copy);
  67 }

  69 int
  70 dbgsysGetLastErrorString(char *buf, int len)
  71 {
  72     const char *s = dlerror();
  73     size_t n;
  74     size_t l = (size_t)len;
  75 
  76     if (len <= 0)
  77         return 0;
  78 
  79     *buf = '\0';
  80     if (s == NULL)
  81         return 0;
  82 
  83     n = strlen(s);
  84     if (n >= l)
  85         n = l - 1;
  86     strncpy(buf, s, n);
  87     buf[n] = '\0'; /* not actually needed */
  88     return n;
  89 }

 104 /*
 105  * create a string for the dynamic lib open call by adding the
 106  * appropriate pre and extensions to a filename and the path
 107  */
 108 void
 109 dbgsysBuildLibName(char *holder, int holderlen, const char *pname,
 110                    const char *fname)
 111 {
 112     const int pnamelen = pname ? strlen(pname) : 0;
 113 
 114     *holder = '\0';
 115     // Quietly truncate on buffer overflow.  Should be an error.
 116     if (pnamelen + (int)strlen(fname) + 10 > holderlen) {
 117         return;
 118     }
 119 
 120     if (pnamelen == 0) {
 121         (void)snprintf(holder, holderlen, "lib%s." LIB_SUFFIX, fname);
 122     } else {
 123       dll_build_name(holder, holderlen, pname, fname);
 124     }
 125 }

 127 void *
 128 dbgsysLoadLibrary(const char *name, char *err_buf, int err_buflen)
 129 {
 130     void * result;
 131     result = dlopen(name, RTLD_LAZY);
 132     if (result == NULL) {
 133         (void)strncpy(err_buf, dlerror(), err_buflen-2);
 134         err_buf[err_buflen-1] = '\0';
 135     }
 136     return result;
 137 }
